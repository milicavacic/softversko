CREATE TABLE Grupa_korisnika(
ID INT AUTO_INCREMENT,
Naziv VARCHAR(50) UNIQUE,
CONSTRAINT Grupa_korisnika_ID_PK PRIMARY KEY(ID)
);

CREATE TABLE Korisnik(
ID INT AUTO_INCREMENT,
Ime VARCHAR(30),
Prezime VARCHAR(30),
UsernameString VARCHAR(30) UNIQUE,
PasswordString VARCHAR(30),
ID_Grupe INT NOT NULL,
Pol TINYINT(1),
Godiste INT,
Visina INT(3),
Masa INT(3),
CONSTRAINT Korisnik_ID_PK PRIMARY KEY(ID),
CONSTRAINT Korisnik_ID_Grupe_FK FOREIGN KEY(ID_Grupe) REFERENCES Grupa_korisnika(ID) 
);


CREATE TABLE Kviz(
ID INT AUTO_INCREMENT,
Pitanje VARCHAR(1000) NOT NULL,
Odgovori VARCHAR(5000) NOT NULL,
Fun_Fact VARCHAR(1000) NOT NULL,
CONSTRAINT Kviz_ID_PK PRIMARY KEY(ID)
);


CREATE TABLE Vezba(
ID INT AUTO_INCREMENT,
Naziv VARCHAR(100) NOT NULL,
Opis VARCHAR(999),
TipVezbe VARCHAR(30),
CONSTRAINT Vezba_ID_PK PRIMARY KEY(ID)
);



CREATE TABLE Sadrzina_mojih_vezbi(
ID INT AUTO_INCREMENT,
ID_Grupa INT NOT NULL,
ID_Vezba INT NOT NULL,
CONSTRAINT Sadrzina_mojih_vezbi_ID_PK PRIMARY KEY(ID),
CONSTRAINT Sadrzina_mojih_vezbi_ID_Grupa_FK FOREIGN KEY(ID_Grupa) REFERENCES Grupa_korisnika(ID),
CONSTRAINT Sadrzina_mojih_vezbi_ID_Vezba_FK FOREIGN KEY(ID_Vezba) REFERENCES Vezba(ID)
);



CREATE TABLE Setovi(
ID INT AUTO_INCREMENT,
Naziv VARCHAR(100) NOT NULL,
Sastojci VARCHAR(200) NOT NULL, 
TipSeta VARCHAR(30) NOT NULL,
ID_Vlasnika INT,
CONSTRAINT Setovi_ID_PK PRIMARY KEY(ID),
CONSTRAINT Setovi_ID_Vlasnika_FK FOREIGN KEY(ID_Vlasnika) REFERENCES Korisnik(ID)
);



CREATE TABLE Setovi_Grupe(
ID INT AUTO_INCREMENT,
ID_Grupa INT NOT NULL,
ID_Setovi INT NOT NULL,
CONSTRAINT Setovi_Grupe_ID_PK PRIMARY KEY(ID),
CONSTRAINT Setovi_Grupe_ID_Grupa_FK FOREIGN KEY(ID_Grupa) REFERENCES Grupa_korisnika(ID),
CONSTRAINT Setovi_Grupe_ID_Setovi_FK FOREIGN KEY(ID_Setovi) REFERENCES Setovi(ID)
);
